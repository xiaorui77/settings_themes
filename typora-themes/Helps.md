# Use

```bash
# night-tao
TYPORA_THEME_DIR="$HOME/Library/Application Support/abnerworks.Typora/themes"

# use
ln -s $MY_GITHUB_HOME/settings_themes/typora-themes/nightly-tao/night-tao $TYPORA_THEME_DIR/night-tao
ln -s $MY_GITHUB_HOME/settings_themes/typora-themes/nightly-tao/night-tao.css $TYPORA_THEME_DIR/night-tao.css

# or
cd $MY_GITHUB_HOME/settings_themes/typora-themes
ln -s nightly-tao/night-tao $TYPORA_THEME_DIR/night-tao
ln -s nightly-tao/night-tao.css $TYPORA_THEME_DIR/night-tao.css

```
